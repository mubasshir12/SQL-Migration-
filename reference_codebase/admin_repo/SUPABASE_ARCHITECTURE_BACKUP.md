# Supabase Project Master Schema & Architecture Backup

This document contains ALL the necessary SQL commands, table schemas, Triggers, and RPC functions required to completely rebuild the current Ceaznet Admin application in a completely blank, new Supabase project. 

If your current Supabase project gets restricted, deleted, or you simply want to deploy this locally/elsewhere, run these scripts **consecutively** from top to bottom in the SQL editor of your new project.

---

## 2. Global Activity Logging & Admin Functions
*This sets up a robust system that listens to ALL tables and creates an audit log for every Insert, Update, and Delete.*

```sql
-- 2.1 The Main Activity Logs Table
CREATE TABLE IF NOT EXISTS public.activity_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    table_name TEXT NOT NULL,
    operation TEXT NOT NULL,
    old_data JSONB,
    new_data JSONB,
    changed_by UUID,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Enable Realtime so the Frontend Activity Feed listens to it
ALTER PUBLICATION supabase_realtime ADD TABLE public.activity_logs;

-- 2.2 The Trigger Function that processes changes
CREATE OR REPLACE FUNCTION public.log_activity_function()
RETURNS TRIGGER AS $$
DECLARE
    v_user_id UUID;
BEGIN
    BEGIN
        v_user_id := (current_setting('request.jwt.claims', true)::jsonb ->> 'sub')::UUID;
    EXCEPTION WHEN OTHERS THEN
        v_user_id := NULL;
    END;

    IF TG_OP = 'INSERT' THEN
        INSERT INTO public.activity_logs (table_name, operation, new_data, changed_by)
        VALUES (TG_TABLE_NAME, TG_OP, row_to_json(NEW)::jsonb, v_user_id);
        RETURN NEW;
    ELSIF TG_OP = 'UPDATE' THEN
        INSERT INTO public.activity_logs (table_name, operation, old_data, new_data, changed_by)
        VALUES (TG_TABLE_NAME, TG_OP, row_to_json(OLD)::jsonb, row_to_json(NEW)::jsonb, v_user_id);
        RETURN NEW;
    ELSIF TG_OP = 'DELETE' THEN
        INSERT INTO public.activity_logs (table_name, operation, old_data, changed_by)
        VALUES (TG_TABLE_NAME, TG_OP, row_to_json(OLD)::jsonb, v_user_id);
        RETURN OLD;
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 2.3 Function to apply triggers to ALL existing tables
CREATE OR REPLACE FUNCTION public.apply_activity_triggers_to_all()
RETURNS void AS $$
DECLARE
    r RECORD;
BEGIN
    FOR r IN SELECT tablename FROM pg_tables WHERE schemaname = 'public' AND tablename != 'activity_logs'
    LOOP
        EXECUTE format('DROP TRIGGER IF EXISTS %I ON public.%I', 'log_activity_' || r.tablename, r.tablename);
        EXECUTE format('CREATE TRIGGER %I AFTER INSERT OR UPDATE OR DELETE ON public.%I FOR EACH ROW EXECUTE FUNCTION public.log_activity_function()', 'log_activity_' || r.tablename, r.tablename);
    END LOOP;
END;
$$ LANGUAGE plpgsql;

-- Apply right now
SELECT public.apply_activity_triggers_to_all();
```

## 3. Intelligent API Key Manager (GNews, Gemini, Brevo)
*We use advanced postgres functions to manage, load-balance, and auto-reset API limits for our background Edge Functions.*

```sql
-- 3.1 Base Keys Table
CREATE TABLE IF NOT EXISTS public.news_api_keys (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    provider TEXT NOT NULL CHECK (provider IN ('gnews', 'gemini', 'brevo')),
    api_key TEXT NOT NULL UNIQUE,
    account_name TEXT,
    status TEXT NOT NULL DEFAULT 'active' CHECK (status IN ('active', 'exhausted')),
    call_count INTEGER NOT NULL DEFAULT 0,
    daily_call_count INTEGER NOT NULL DEFAULT 0,
    failure_count INTEGER NOT NULL DEFAULT 0,
    last_reset_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 3.2 Daily Usage Metrics Table
CREATE TABLE IF NOT EXISTS public.api_key_daily_usage (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    api_key_id UUID NOT NULL REFERENCES public.news_api_keys(id) ON DELETE CASCADE,
    provider TEXT NOT NULL,
    usage_date DATE NOT NULL,
    call_count INTEGER NOT NULL DEFAULT 0,
    failure_count INTEGER NOT NULL DEFAULT 0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    UNIQUE(api_key_id, usage_date)
);

-- 3.3 Trigger for auto-updated timestamps
CREATE OR REPLACE FUNCTION public.update_news_api_keys_updated_at()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = now();
    RETURN NEW;
END;
$$ language 'plpgsql';

CREATE TRIGGER update_news_api_keys_updated_at
    BEFORE UPDATE ON public.news_api_keys
    FOR EACH ROW EXECUTE FUNCTION public.update_news_api_keys_updated_at();

-- 3.4 Provider specific timezone checker for Limits Unlocking (Lazy Reset)
CREATE OR REPLACE FUNCTION public.get_provider_current_date(p_provider TEXT) 
RETURNS DATE AS $$
BEGIN
    IF p_provider = 'gemini' THEN RETURN (now() AT TIME ZONE 'America/Los_Angeles')::DATE;
    ELSE RETURN (now() AT TIME ZONE 'UTC')::DATE;
    END IF;
END;
$$ LANGUAGE plpgsql STABLE;

CREATE OR REPLACE FUNCTION public.needs_daily_reset(p_provider TEXT, p_last_reset_at TIMESTAMPTZ) 
RETURNS BOOLEAN AS $$
DECLARE
    v_last_date DATE;
    v_curr_date DATE;
BEGIN
    IF p_provider = 'gemini' THEN
        v_last_date := (p_last_reset_at AT TIME ZONE 'America/Los_Angeles')::DATE;
        v_curr_date := (now() AT TIME ZONE 'America/Los_Angeles')::DATE;
    ELSE
        v_last_date := (p_last_reset_at AT TIME ZONE 'UTC')::DATE;
        v_curr_date := (now() AT TIME ZONE 'UTC')::DATE;
    END IF;
    RETURN v_curr_date > v_last_date;
END;
$$ LANGUAGE plpgsql STABLE;

CREATE OR REPLACE FUNCTION public.perform_lazy_daily_reset(p_provider TEXT)
RETURNS VOID AS $$
BEGIN
    UPDATE public.news_api_keys
    SET daily_call_count = 0, failure_count = 0, status = 'active', last_reset_at = now(), updated_at = now()
    WHERE provider = p_provider AND public.needs_daily_reset(provider, last_reset_at) = true;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 3.5 Retrieve API Key Function (Used by Edge Functions to fetch keys via RPC)
CREATE OR REPLACE FUNCTION public.get_active_news_api_key(p_provider TEXT)
RETURNS TEXT AS $$
DECLARE
    v_api_key TEXT; v_id UUID; v_curr_date DATE;
BEGIN
    PERFORM public.perform_lazy_daily_reset(p_provider);
    SELECT id, api_key INTO v_id, v_api_key FROM public.news_api_keys
    WHERE provider = p_provider AND status = 'active'
    ORDER BY daily_call_count ASC, updated_at ASC LIMIT 1 FOR UPDATE SKIP LOCKED;

    IF v_id IS NOT NULL THEN
        UPDATE public.news_api_keys SET call_count = call_count + 1, daily_call_count = daily_call_count + 1, updated_at = now() WHERE id = v_id;
        v_curr_date := public.get_provider_current_date(p_provider);
        INSERT INTO public.api_key_daily_usage (api_key_id, provider, usage_date, call_count)
        VALUES (v_id, p_provider, v_curr_date, 1) ON CONFLICT (api_key_id, usage_date) 
        DO UPDATE SET call_count = public.api_key_daily_usage.call_count + 1, provider = EXCLUDED.provider;
        RETURN v_api_key;
    ELSE RETURN NULL;
    END IF;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```

## 4. Edge Functions Observability & Logging
*These tables store output from Edge Functions so we can render "Database Analytics" directly in the frontend Settings page.*

```sql
-- 4.1 Log Tables
CREATE TABLE IF NOT EXISTS update_news_logs (
    id BIGSERIAL PRIMARY KEY,
    status TEXT NOT NULL,
    error_message TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS enrich_sources_logs (
    id BIGSERIAL PRIMARY KEY, status TEXT NOT NULL, error_message TEXT, created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS image_proxy_logs (
    id BIGSERIAL PRIMARY KEY, status TEXT NOT NULL, error_message TEXT, created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 4.2 Aggregated Analytics Function (Fetched by Dashboard GUI)
CREATE OR REPLACE FUNCTION get_function_stats()
RETURNS TABLE(function_name text, total_calls bigint, success_count bigint, error_count bigint, last_run timestamptz) AS $$
BEGIN
  RETURN QUERY
  SELECT 'update-news'::text as function_name, COUNT(*) as total_calls, COUNT(*) FILTER (WHERE status = 'SUCCESS') as success_count, COUNT(*) FILTER (WHERE status != 'SUCCESS') as error_count, MAX(created_at) as last_run FROM update_news_logs
  UNION ALL
  SELECT 'enrich-sources'::text as function_name, COUNT(*) as total_calls, COUNT(*) FILTER (WHERE status = 'SUCCESS') as success_count, COUNT(*) FILTER (WHERE status != 'SUCCESS') as error_count, MAX(created_at) as last_run FROM enrich_sources_logs
  UNION ALL
  SELECT 'image-proxy'::text as function_name, COUNT(*) as total_calls, COUNT(*) FILTER (WHERE status = 'SUCCESS') as success_count, COUNT(*) FILTER (WHERE status != 'SUCCESS') as error_count, MAX(created_at) as last_run FROM image_proxy_logs;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```

## 5. Security - Row Level Security (RLS) Passers
*Bypasses regular blocks for Administrative Access within the context of the React App.*

```sql
ALTER TABLE public.news_api_keys ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.api_key_daily_usage ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.news_system_config ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.news_articles ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Admin Full Access" ON public.news_api_keys FOR ALL TO authenticated USING (true) WITH CHECK (true);
CREATE POLICY "Admin Full Access" ON public.api_key_daily_usage FOR ALL TO authenticated USING (true) WITH CHECK (true);
CREATE POLICY "Admin Full Access" ON public.news_system_config FOR ALL TO authenticated USING (true) WITH CHECK (true);
CREATE POLICY "Admin Full Access" ON public.news_articles FOR ALL TO authenticated USING (true) WITH CHECK (true);
```
